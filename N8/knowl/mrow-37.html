\begin{align*} f(x)\amp= a^x\amp \ct{Tomando o $\log$ em ambos os lados}\\ \ln f(x)\amp= x \ln a\amp \ct{Exponenciar na base $e$ em ambos os lados}\\ f(x)\amp= e^{x \ln a}\qquad \qquad\qquad \qquad \ct{Regra da Cadeia}\\ f'(x)\amp= e^{x \ln{a}} \cdot \ln{a}\\ \amp= a^x \cdot \ln{a} \end{align*}
