Use a Regra da Cadeia com as regras para funções trigonométricas inversas.

\begin{equation*} \dfrac{d}{dt}\left[ 10 \sin^{-1} 5 t \right] = 10 \cdot \frac{1} {\sqrt{1 - (5 t)^2}} \cdot 5 = \frac{50}{\sqrt{1 - 25 t^2}}. \end{equation*}
\begin{equation*} \dfrac{d}{dx}\left[ 8 \cos^{-1} 5 x \right] = 8 \cdot -\frac{1} {\sqrt{1 - (5 x)^2}} \cdot 5 = - \frac{40}{\sqrt{1 - 25 x^2}}. \end{equation*}
\begin{equation*} \dfrac{d}{du}\left[ 6 \sec^{-1} 4 u \right] = 6 \cdot \frac{1} {|4 u|\sqrt{(4 u)^2 - 1}} \cdot 4 = \frac{6}{|u|\sqrt{16 u^2 - 1}}. \end{equation*}
\begin{equation*} \dfrac{d}{dz}\left[ 10 \tan^{-1} 3 z \right] = 10 \cdot \frac{1} {1 + (3 z)^2} \cdot 3 = \frac{30}{1 + 9 z^2}. \end{equation*}
\begin{equation*} \dfrac{d}{dy}\left[ 9 \csc^{-1} 4 y \right] = 9 \cdot -\frac{1} {|4 y|\sqrt{(4 y)^2 - 1}} \cdot 4 = -\frac{9}{|y|\sqrt{16 y^2 - 1}}. \end{equation*}
\begin{equation*} \dfrac{d}{dq}\left[ 10 \tan^{-1} 3 z \right] = 11 \cdot -\frac{1} {1 + (5 q)^2} \cdot 5 = -\frac{55}{1 + 25 q^2}. \end{equation*}
