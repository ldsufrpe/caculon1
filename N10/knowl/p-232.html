\begin{align*} f(x) \amp= \log x \amp g(x) \amp=\tan\Big(\frac{\pi}{2}-x\Big)\\ f'(x)\amp=\frac{1}{x}\qquad\amp g'(x)\amp=-\sec^2\Big(\frac{\pi}{2}-x\Big) \end{align*}

and the second time with

\begin{align*} f(x) \amp= \cos^2\Big(\frac{\pi}{2}-x\Big) \amp g(x) \amp=x\\ f'(x)\amp=2\cos\Big(\frac{\pi}{2}-x\Big) \Big[-\sin\Big(\frac{\pi}{2}-x\Big)\Big](-1)\qquad\amp g'(x)\amp=1 \end{align*}
