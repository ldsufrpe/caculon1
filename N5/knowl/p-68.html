\begin{align*} f'(x)=\amp \lim\limits_{h\to 0}\frac{f(x+h)-f(x)}{h}\\ =\amp \lim\limits_{h\to 0}\frac{\sqrt{x+h}-\sqrt{x}}{h}\\ =\amp \lim\limits_{h\to 0}\left(\frac{\sqrt{x+h}-\sqrt{x}}{h}\cdot \frac{\sqrt{x+h}+\sqrt{x}}{\sqrt{x+h}+\sqrt{x}}\right)\\ = \amp\lim\limits_{h\to 0}\frac{(x+h)-x}{h(\sqrt{x+h}+\sqrt{x})}\\ = \amp \lim\limits_{h\to 0}\frac{\cancel{h}}{\cancel{h}(\sqrt{x+h}+\sqrt{x})}\\ = \amp \lim\limits_{h\to 0}\frac{1}{\sqrt{x}+\sqrt{x}}=\frac{1}{2\sqrt{x}}. \end{align*}
